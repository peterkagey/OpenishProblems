\begin{question}
  What is the probability of seeing a $k$-gon (for some fixed $k$), when given
  a uniformly random function $f_{n,m}$?
\end{question}

\begin{related}
  \item What if $f_{n,n}$ is restricted to be a permutation?
  \item What if $f_{n,m}$ is injective?
\end{related}
